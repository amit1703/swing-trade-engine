## Background

### How `_patch_constants` works

The optimizer temporarily overrides module-level constants before each WFO run:

```python
@contextmanager
def _patch_constants(params):
    _preload_modules()          # ensures modules are in sys.modules
    saved = []
    for param_key, patches in _MODULE_PATCHES.items():
        val = params[param_key]
        for mod_name, attr in patches:
            mod = sys.modules[mod_name]
            saved.append((mod, attr, getattr(mod, attr)))
            setattr(mod, attr, val)       # ← overrides the module attribute
    try:
        yield
    finally:
        for mod, attr, orig in saved:
            setattr(mod, attr, orig)      # ← restores on exit
```

`filters.compute_regime_series()` reads `REGIME_SELECTIVE_THRESHOLD` as a bare name on its last line:
`return score >= REGIME_SELECTIVE_THRESHOLD`

Python resolves bare names from the module's `__dict__` at call time, so `setattr(filters_module, "REGIME_SELECTIVE_THRESHOLD", 30)` will be seen by the next `compute_regime_series()` call. ✓

`_preload_modules()` iterates `_MODULE_PATCHES` automatically — adding `("filters", ...)` to the patches dict is sufficient to ensure `filters` gets imported before patching.

### What `REGIME_BULL_THRESHOLD` controls

Maps to `filters.REGIME_SELECTIVE_THRESHOLD` (currently 40/100 points).

- **Low value (20):** Trade in almost any SPY regime — permissive
- **High value (55):** Only trade when SPY is clearly bullish — restrictive
- Search range `20–55` stays safely below the max possible SPY-only score (~60 pts) so the gate doesn't permanently block all trading

def _export_best(study, suppress_output: bool = False) -> None:
    """Print summary and write config/best_parameters.json."""
    import optuna.importance                              # ← add this import

    best = study.best_trial
    metrics = best.user_attrs.get("metrics", {})

    # Compute parameter importance (FAnova method; silently skip if < 2 trials)
    param_importance: dict = {}
    try:
        param_importance = {
            k: round(v, 6)
            for k, v in optuna.importance.get_param_importances(study).items()
        }
    except Exception:
        pass

    output = {
        "generated_at":    datetime.now(timezone.utc).isoformat(),
        "study_name":      study.study_name,
        "best_trial":      best.number,
        "best_score":      round(best.value, 6),
        "parameters":      {
            k: round(v, 6) if isinstance(v, float) else v
            for k, v in best.params.items()
        },
        "oos_metrics":     metrics,
        "param_importance": param_importance,            # ← add this
    }

    _OUTPUT_PATH.parent.mkdir(parents=True, exist_ok=True)
    _OUTPUT_PATH.write_text(json.dumps(output, indent=2))

    if not suppress_output:
        print("\n" + "="*55)
        print("  BEST PARAMETERS")
        print("="*55)
        for k, v in best.params.items():
            print(f"  {k:<25} {round(v, 4) if isinstance(v, float) else v}")
        print("\n  OOS Performance:")
        for k, v in metrics.items():
            print(f"  {k:<25} {v}")
        if param_importance:
            print("\n  Parameter Importance:")
            for k, v in sorted(param_importance.items(), key=lambda x: -x[1]):
                bar = "█" * max(1, int(v * 30))
                print(f"  {k:<25} {v:.4f}  {bar}")
        print(f"\n  Robustness Score:  {best.value:.4f}")
        print(f"  Exported to:       {_OUTPUT_PATH}")
        print("="*55)
```

# ── Diagnostics log ───────────────────────────────────────────────────────────
_LOG_PATH = _PROJECT_DIR / "optuna_trial_log.csv"


def _log_trial(
    trial_num: int,
    score: float,
    params: dict[str, Any],
    metrics: dict[str, Any],
) -> None:
    """Append one trial's key stats to a CSV log for live monitoring."""
    header = not _LOG_PATH.exists()
    param_keys = [
        "ATR_MULTIPLIER", "VCP_TIGHTNESS_RANGE", "BREAKOUT_BUFFER_ATR",
        "BREAKOUT_VOL_MULT", "TARGET_RR", "TRAIL_ATR_MULT", "REGIME_BULL_THRESHOLD",
    ]
    try:
        with open(_LOG_PATH, "a") as f:
            if header:
                cols = ["trial", "score", "profit_factor", "expectancy",
                        "max_drawdown_pct", "total_trades", "win_rate"] + param_keys
                f.write(",".join(cols) + "\n")
            row = [
                trial_num,
                round(score, 6),
                round(metrics.get("profit_factor", 0), 4),
                round(metrics.get("expectancy", 0), 4),
                round(metrics.get("max_drawdown_pct", 0), 2),
                metrics.get("total_trades", 0),
                round(metrics.get("win_rate", 0), 2),
            ] + [round(params.get(k, 0), 6) for k in param_keys]
            f.write(",".join(str(v) for v in row) + "\n")
    except Exception:
        pass  # Never crash the optimizer over a log write
```
