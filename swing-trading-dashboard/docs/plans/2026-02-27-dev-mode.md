# Dev Mode Implementation Plan

> **For Claude:** REQUIRED SUB-SKILL: Use superpowers:executing-plans to implement this plan task-by-task.

**Goal:** Add a toggleable Dev Mode overlay with halt-bypass scanning, engine health metrics, dry-run mode, and a per-ticker debug drill-down panel.

**Architecture:** `devMode` boolean lives in App.jsx React state only (stateless, resets on refresh). Frontend appends `?force=true` / `?dry_run=true` query params to the scan trigger. Backend `_run_scan()` gains `force` and `dry_run` kwargs; the halt gate becomes `if not regime["is_bullish"] and not force`. Per-engine counters added to `_scan_state["engine_stats"]` and exposed via the existing `/api/scan-status` endpoint. A new `GET /api/debug/{ticker}` endpoint runs all engines live and returns structured JSON. Three new frontend components: `EngineHealthPanel.jsx`, `DebugDrawer.jsx`, and a `[DEV]` toggle wired into `Header.jsx`.

**Tech Stack:** FastAPI + Python backend, React 18 + Vite frontend, existing pytest suite.

```python
async def trigger_scan(
    background_tasks: BackgroundTasks,
    force: bool = Query(False, description="Bypass bearish halt gate"),
    dry_run: bool = Query(False, description="Run pipeline without saving to DB"),
):
    if _scan_state["in_progress"]:
        return {
            "status": "already_running",
            "progress": _scan_state["progress"],
            "total": _scan_state["total"],
        }

    scan_ts = datetime.utcnow().strftime("%Y-%m-%dT%H:%M:%S")
    background_tasks.add_task(_run_scan, scan_ts, ACTIVE_UNIVERSE, force, dry_run)

    return {
        "status": "started",
        "scan_timestamp": scan_ts,
        "tickers": len(ACTIVE_UNIVERSE),
        "forced": force,
        "dry_run": dry_run,
        "message": f"Scanning {len(ACTIVE_UNIVERSE)} tickers in background",
```

```python
@app.get("/api/debug/{ticker}")
async def debug_ticker(ticker: str):
    """
    Dev mode: run all engines live for a single ticker and return
    structured pass/fail results for the DebugDrawer component.
    """
    sym = ticker.upper()
    loop = asyncio.get_event_loop()

    # Fetch OHLCV
    df = await _fetch(sym)
    if df is None or df.empty:
        raise HTTPException(status_code=404, detail=f"No data for {sym}")
    if len(df) < MIN_CANDLES_FOR_ANALYSIS:
        raise HTTPException(status_code=422, detail=f"Insufficient history for {sym}")

    # Indicators for display
    adj = "Adj Close" if "Adj Close" in df.columns else "Close"
    close_adj = df[adj]
    ema8_s  = _ema(close_adj, 8)
    ema20_s = _ema(close_adj, 20)
    sma50_s = _sma(close_adj, 50)
    cci20_s = _cci(df["High"], df["Low"], close_adj, 20)

    lc   = float(close_adj.iloc[-1])
    l8   = float(ema8_s.iloc[-1])   if pd.notna(ema8_s.iloc[-1])  else None
    l20  = float(ema20_s.iloc[-1])  if pd.notna(ema20_s.iloc[-1]) else None
    l50  = float(sma50_s.iloc[-1])  if pd.notna(sma50_s.iloc[-1]) else None
    lcci = float(cci20_s.iloc[-1])  if pd.notna(cci20_s.iloc[-1]) else None

    # Zones
    zones = await get_sr_zones_for_ticker_from_db(DB_PATH, sym)
    if not zones:
        try:
            zones = await loop.run_in_executor(None, calculate_sr_zones, sym, df)
        except Exception:
            zones = []

    # RS metrics
    rs_ratio     = 0.0
    rs_blue_dot  = False
    rs_score     = 0
    try:
        spy_df = await _fetch("SPY")
        if spy_df is not None and len(spy_df) >= MIN_CANDLES_FOR_RS:
            rs_line = await loop.run_in_executor(None, calculate_rs_line, df, spy_df)
            if rs_line and len(rs_line) >= MIN_CANDLES_FOR_RS:
                rs_stats = get_rs_stats(rs_line)
                rs_ratio    = round(float(rs_stats.get("rs_ratio", 0.0)), 4)
                rs_blue_dot = bool(detect_rs_blue_dot(rs_line))
                rs_score    = int(await loop.run_in_executor(None, calculate_rs_score, df, spy_df))
    except Exception as exc:
        log.warning("debug RS failed for %s: %s", sym, exc)

    # Regime
    regime = await get_latest_regime(DB_PATH) or {
        "is_bullish": False, "spy_close": 0.0, "spy_20ema": 0.0, "regime": "NO_DATA"
    }

    # Run engines — capture triggered/signal
    def _run_engine(fn, *args):
        try:
            result = fn(*args)
            return result
        except Exception:
            return None

    e2 = await loop.run_in_executor(
        None, _run_engine, scan_vcp, sym, df, zones,
        0.0, rs_ratio, 0.0, rs_blue_dot, rs_score
    )
    e3 = await loop.run_in_executor(
        None, _run_engine, scan_pullback, sym, df, zones, None
    )
    if e3 is None:
        e3 = await loop.run_in_executor(
            None, _run_engine, scan_relaxed_pullback, sym, df, zones, None
        )
    e5 = await loop.run_in_executor(
        None, _run_engine, scan_base_pattern, sym, df, rs_ratio, rs_score, rs_blue_dot
    )
    e6 = await loop.run_in_executor(
        None, _run_engine, scan_resistance_breakout, sym, df, zones
    ) if zones else None

    def _eng(result, extra_keys=()):
        if result is None:
            return {"triggered": False, "signal": None}
        out = {"triggered": True, "signal": result.get("setup_type") or result.get("signal")}
        for k in extra_keys:
            if k in result:
                out[k] = result[k]
        return out

    return {
        "ticker": sym,
        "regime": {
            "is_bullish": regime.get("is_bullish"),
            "spy_close":  regime.get("spy_close"),
            "spy_20ema":  regime.get("spy_20ema"),
        },
        "indicators": {
            "close": round(lc, 2),
            "ema8":  round(l8, 2)  if l8  else None,
            "ema20": round(l20, 2) if l20 else None,
            "sma50": round(l50, 2) if l50 else None,
            "cci":   round(lcci, 1) if lcci else None,
            "above_ema8":  bool(l8  and lc > l8),
            "above_ema20": bool(l20 and lc > l20),
            "above_sma50": bool(l50 and lc > l50),
            "rs_ratio":    rs_ratio,
            "rs_blue_dot": rs_blue_dot,
            "rs_score":    rs_score,
        },
        "zones": zones,
        "engine2": _eng(e2, ("is_breakout", "is_vol_surge", "path", "is_rs_lead")),
        "engine3": _eng(e3, ("is_relaxed",)),
        "engine5": _eng(e5, ("base_type", "quality_score")),
        "engine6": _eng(e6, ("days_since_breakout", "volume_ratio")),
    }
```
